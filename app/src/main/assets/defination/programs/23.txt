
public class ForLoop{

    public static void main(String[] args) {
        //learn
        for (int i = 0; i < 10; i++) {
            System.out.println(i);
        }
        System.out.println();
        //explore
        char a = 'A';
        for (int i = 0; i < 10; i++) {
            System.out.println(a);
            a++;
        }
    }
}
