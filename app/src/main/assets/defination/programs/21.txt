
public class WhileLoop{

    public static void main(String[] args) {

        //learning
        int a = 1;
        while (a <= 10) {
            System.out.println(a);
            a++;
        }
        //exploring
        int sum = 0, b = 1;
        while (b <= 5) {
            sum += b;
            b++;
        }
        System.out.println("Sum of 1-5 is " + sum);

    }
}
